\begin{frame}
  \frametitle{HSA (Dynamic Effect): Equation}
  % Equation
  \begin{equation*}
    \begin{aligned}
      \pi_t &= \alpha \pi_{t-1} + (1-\alpha)\mathbb{E}_t \pi_{t+1} + \kappa x_t {\color{red!70!black}{- \theta \hat{N}_t}} + e_t \\
      x_t &= \phi_1 x_{t-1} + \zeta_t \\
      N_t &= \hat{N}_t + \bar{N}_t \\
      \hat{N}_t &= \rho_1 \hat{N}_{t-1} + \rho_2 \hat{N}_{t-2} + u_t \\
      \bar{N}_t &= n + \bar{N}_{t-1} + \epsilon_t
    \end{aligned}
  \end{equation*}
  % Shock structure
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \centering
      \scriptsize
      \begin{equation*}
        \begin{bmatrix}
          e_t \\ \zeta_t \\ u_t \\ \epsilon_t
        \end{bmatrix}
        \sim N\!\left(
        0,
        \begin{bmatrix}
          \sigma_e^2 & \sigma_{e\zeta} & 0 & 0 \\
          \sigma_{e\zeta} & \sigma_\zeta^2 & 0 & 0 \\
          0 & 0 & \sigma_u^2 & 0 \\
          0 & 0 & 0 & \sigma_\epsilon^2
        \end{bmatrix}
        \right)
      \end{equation*}
    \end{column}

    \begin{column}{0.48\textwidth}
